import java.util.Scanner;
public class MultiplicationTable {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        System.out.print("Enter a number: ");
        int num = scanner.nextInt(); // Take user input
       System.out.println("Multiplication Table for " + num);
        for (int i = 1; i <= 10; i++) {
            System.out.println(num + " * " + i + " = " + (num * i));
        }
       scanner.close();
    }
}

import java.util.LinkedList;
import java.util.Iterator;
import java.util.ListIterator;
public class ColorLinkedList {
    public static void main(String[] args) {
        LinkedList<String> colorList = new LinkedList<>();
        colorList.add("Red");
        colorList.add("Blue");
        colorList.add("Yellow");
        colorList.add("Orange");
        System.out.println("Original List:");
        Iterator<String> iterator = colorList.iterator();
        while (iterator.hasNext()) {
            System.out.println(iterator.next());
        }
       System.out.println("\nList in reverse order:");
        ListIterator<String> listIterator = colorList.listIterator(colorList.size());
        while (listIterator.hasPrevious()) {
            System.out.println(listIterator.previous());
        }
       LinkedList<String> newColors = new LinkedList<>();
        newColors.add("Pink");
        newColors.add("Green");
       int yellowIndex = colorList.indexOf("Yellow");
        colorList.addAll(yellowIndex, newColors);
       System.out.println("\nList after inserting Pink and Green between Blue and Yellow:");
        for (String color : colorList) {
            System.out.println(color);
        }
    }
}
