class MyNumber {
    private int number;
   public MyNumber(int number) {
        this.number = number;
    }
   public boolean isOdd() {
        return number % 2 != 0;
    }
   public static void main(String[] args) {
        if (args.length > 0) {
            try {
                int num = Integer.parseInt(args[0]);
                MyNumber myNum = new MyNumber(num);
                if (myNum.isOdd()) {
                    System.out.println(num + " is an odd number.");
                } else {
                    System.out.println(num + " is an even number.");
                }
            } catch (NumberFormatException e) {
                System.out.println("Please enter a valid integer.");
            }
        } else {
            System.out.println("Please provide a number as a command-line argument.");
        }
    }
}

class Student {
    private int rollno;
    private String name;
    private static int objectCount = 0;
    public Student() {
        this.rollno = 0;
        this.name = "Unknown";
        objectCount++; // Increment object count
        displayCount(); // Display object count after creation
    }
   public Student(int rollno, String name) {
        this.rollno = rollno;
        this.name = name;
        objectCount++; // Increment object count
        displayCount(); // Display object count after creation
    }
   public static void displayCount() {
        System.out.println("Total number of Student objects created: " + objectCount);
    }
   public void displayStudentDetails() {
        System.out.println("Roll No: " + rollno);
        System.out.println("Name: " + name);
    }
   public static void main(String[] args) {
        Student student1 = new Student(101, "Sakshi");
        student1.displayStudentDetails();
        Student student2 = new Student(102, "Sohail");
        student2.displayStudentDetails();
       Student student3 = new Student(103, "Jasmin");
        student3.displayStudentDetails();
    }
}
