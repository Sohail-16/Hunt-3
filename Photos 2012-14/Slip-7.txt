import java.util.Hashtable;
import java.util.Scanner;
import java.util.Map;
public class EmployeeHashTable {
    public static void main(String[] args) {
        Hashtable<String, Double> employeeTable = new Hashtable<>();
       Scanner scanner = new Scanner(System.in);
        System.out.print("Enter the number of employees: ");
        int n = scanner.nextInt();
        scanner.nextLine(); // Consume newline character
       for (int i = 0; i < n; i++) {
            System.out.println("Enter details for Employee " + (i + 1) + ":");
            System.out.print("Employee Name: ");
            String name = scanner.nextLine();
            System.out.print("Salary: ");
            double salary = scanner.nextDouble();
            scanner.nextLine(); // Consume newline character
           employeeTable.put(name, salary);
        }
       System.out.println("\nEmployee Details:");
        for (Map.Entry<String, Double> entry : employeeTable.entrySet()) {
            System.out.println("Name: " + entry.getKey() + ", Salary: " + entry.getValue());
        }
       scanner.close();
    }
}

import java.util.Scanner;
class Student {
    private int rollno;
    private String name;
    private double percentage;
    public Student() {
        this.rollno = 0;
        this.name = "";
        this.percentage = 0.0;
    }
    public Student(int rollno, String name, double percentage) {
        this.rollno = rollno;  
        this.name = name;
        this.percentage = percentage;
    }
    public void display() {
        System.out.println("Roll No: " + this.rollno);
        System.out.println("Name: " + this.name);
        System.out.println("Percentage: " + this.percentage);
        System.out.println("----------------------------");
    }
}
public class StudentDemo {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
       Student[] students = new Student[5];
       for (int i = 0; i < 3; i++) {
            System.out.println("Enter details for Student " + (i + 1) + ":");
            System.out.print("Roll No: ");
            int rollno = scanner.nextInt();
            scanner.nextLine(); // Consume newline character
           System.out.print("Name: ");
            String name = scanner.nextLine();
           System.out.print("Percentage: ");
            double percentage = scanner.nextDouble();
            scanner.nextLine(); // Consume newline character
           students[i] = new Student(rollno, name, percentage);
        }
       System.out.println("\nDisplaying Student Details:");
        for (Student student : students) {
            student.display();
        }
       scanner.close();
    }
}

import javax.swing.*;
import java.awt.*;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;

public class CustomerAccountDetails extends JFrame implements ActionListener {

    // Components of the form
    private JTextField customerNameField, bankNameField, accountNoField, panNoField;
    private JButton submitButton;

    // Constructor to set up the form
    public CustomerAccountDetails() {
        // Set the title of the form
        setTitle("Customer Account Details");

        // Set the size and layout of the form
        setSize(400, 300);
        setLayout(new GridLayout(5, 2, 10, 10));  // 5 rows, 2 columns, spacing of 10px

        // Create the labels and text fields
        JLabel customerNameLabel = new JLabel("Name of Customer:");
        customerNameField = new JTextField();

        JLabel bankNameLabel = new JLabel("Name of Bank:");
        bankNameField = new JTextField();

        JLabel accountNoLabel = new JLabel("Account No.:");
        accountNoField = new JTextField();

        JLabel panNoLabel = new JLabel("Pan Number:");
        panNoField = new JTextField();

        // Create the submit button
        submitButton = new JButton("Submit");
        submitButton.addActionListener(this);

        // Add components to the frame
        add(customerNameLabel);
        add(customerNameField);
        add(bankNameLabel);
        add(bankNameField);
        add(accountNoLabel);
        add(accountNoField);
        add(panNoLabel);
        add(panNoField);
        add(new JLabel("")); // Empty label for spacing
        add(submitButton);

        // Set the default close operation
        setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);

        // Set the form to be visible
        setVisible(true);
    }

    // Action performed method to handle submit button click
    @Override
    public void actionPerformed(ActionEvent e) {
        if (e.getSource() == submitButton) {
            // Fetch the input values
            String customerName = customerNameField.getText();
            String bankName = bankNameField.getText();
            String accountNo = accountNoField.getText();
            String panNo = panNoField.getText();

            // Display the customer account details in a dialog box
            JOptionPane.showMessageDialog(this,
                    "Customer Name: " + customerName +
                            "\nBank Name: " + bankName +
                            "\nAccount No.: " + accountNo +
                            "\nPAN Number: " + panNo,
                    "Account Details",
                    JOptionPane.INFORMATION_MESSAGE);
        }
    }

    // Main method to run the application
    public static void main(String[] args) {
        new CustomerAccountDetails();
    }
}
